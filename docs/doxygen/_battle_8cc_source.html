59 void Battle::Update() {
60  Ally* player = Manager::getInstance()->player_;
61 
62  if (click_) {
63  if (this->is_over_) {
64  //Restore player attack and defense stats
65  Manager::getInstance()->player_->attack_ = original_attack_;
66  Manager::getInstance()->player_->defense_ = original_defense_;
67 
68  //Notice that the player's level may increase several times in a single battle
69  while (player->xp_ >= player->next_level_xp_) {
70  player->xp_ = player->xp_ - player->next_level_xp_;
71  player->LevelUp();
72  }
73  //Go to the GameWon screen if this was the final boss
74  if (enemy_->name_ == "Black_Dragon" && player->HP_ > 0 && enemy_->HP_ <= 0) {
75  delete Manager::getInstance()->screen_;
76  Manager::getInstance()->screen_ = new GameWon();
77  //Return to game otherwise
78  } else {
79  delete Manager::getInstance()->screen_;
80  Manager::getInstance()->screen_ = new Game();
81  }
82  } else {
83  int clicked_button = CheckButtonsClick();
84  switch (clicked_button) {
85  case 0:
86  Fight(true);
87  break;
88  case 1:
89  drawing_spells_ = !drawing_spells_;
90  break;
91  case 2:
92  if (player->num_healing_potions_ > 0) {
93  player->num_healing_potions_--;
94  log_ += "You use a healing potion and gain 15 HP\n";
95  player->HP_ += 15;
96  if (player->HP_ > player->max_HP_) {
97  player->HP_ = player->max_HP_;
98  }
99  } else {
100  log_ += "You do not have any potions left!\n";
101  }
102  //Potion
103  break;
104  case 3:
105  Flee();
106  break;
107  case 4:
108  case 5:
109  case 6:
110  if (player->spells_[clicked_button-4]->Cast(player, enemy_)) {
111  log_ += player->spells_[clicked_button-4]->log_line_;
112  Fight(false);
113  } else {
114  log_ += "You do not have enough mana!\n";
115  }
116  break;
117  default:
118  break;
119  }
