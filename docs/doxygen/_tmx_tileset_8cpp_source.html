99  void Tileset::Parse(const tinyxml2::XMLNode *tilesetNode, const std::string& file_path)
100  {
101  const tinyxml2::XMLElement *tilesetElem = tilesetNode->ToElement();
102 
103  // Read all the attributes into local variables.
104 
105  // The firstgid and source attribute are kept in the TMX map,
106  // since they are map specific.
107  first_gid = tilesetElem->IntAttribute("firstgid");
108 
109  // If the <tileset> node contains a 'source' tag,
110  // the tileset config should be loaded from an external
111  // TSX (Tile Set XML) file. That file has the same structure
112  // as the <tileset> element in the TMX map.
113  const char* source_name = tilesetElem->Attribute("source");
114  tinyxml2::XMLDocument tileset_doc;
115  if ( source_name )
116  {
117  std::string fileName = file_path + source_name;
118  tileset_doc.LoadFile( fileName.c_str() );
119 
120  if ( tileset_doc.ErrorID() != 0)
121  {
122  fprintf(stderr, "failed to load tileset file '%s'\n", fileName.c_str());
123  return;
124  }
125 
126  // Update node and element references to the new node
127  tilesetNode = tileset_doc.FirstChildElement("tileset");
128  tilesetElem = tilesetNode->ToElement();
129  }
130 
131  tile_width = tilesetElem->IntAttribute("tilewidth");
132  tile_height = tilesetElem->IntAttribute("tileheight");
133  margin = tilesetElem->IntAttribute("margin");
134  spacing = tilesetElem->IntAttribute("spacing");
135  name = tilesetElem->Attribute("name");
136 
137  // Parse the tile offset, if it exists.
138  const tinyxml2::XMLNode *tileOffsetNode = tilesetNode->FirstChildElement("tileoffset");
139  if (tileOffsetNode)
140  {
141  tileOffset = new TileOffset();
142  tileOffset->Parse(tileOffsetNode);
143  }
144 
145  // Parse the terrain types if any.
146  const tinyxml2::XMLNode *terrainTypesNode = tilesetNode->FirstChildElement("terraintypes");
147  if (terrainTypesNode)
148  {
149  TerrainArray terrainArray;
150  terrainArray.Parse(&terrainTypes, terrainTypesNode);
151  }
152 
153  // Parse the image.
154  const tinyxml2::XMLNode *imageNode = tilesetNode->FirstChildElement("image");
155  if (imageNode)
156  {
157  image = new Image();
158  image->Parse(imageNode);
159  }
160 
161  // Iterate through all of the tile elements and parse each.
162  const tinyxml2::XMLNode *tileNode = tilesetNode->FirstChildElement("tile");
163  for (int tId = 0; tileNode; ++tId)
164  {
165  Tile* tile = new Tile(tId);
166  tile->Parse(tileNode);
167  tiles.push_back(tile);
168 
169  tileNode = tileNode->NextSiblingElement("tile");
170  }
171 
172  // Parse the properties if any.
173  const tinyxml2::XMLNode *propertiesNode = tilesetNode->FirstChildElement("properties");
174  if (propertiesNode)
175  {
176  properties.Parse(propertiesNode);
177  }
178  }
