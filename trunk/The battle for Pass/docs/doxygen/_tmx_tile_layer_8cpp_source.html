127  void TileLayer::ParseXML(const tinyxml2::XMLNode *dataNode)
128  {
129  const tinyxml2::XMLNode *tileNode = dataNode->FirstChildElement("tile");
130  int tileCount = 0;
131 
132  while (tileNode)
133  {
134  const tinyxml2::XMLElement *tileElem = tileNode->ToElement();
135 
136  unsigned gid = 0;
137 
138  // Read the Global-ID of the tile.
139  const char* gidText = tileElem->Attribute("gid");
140 
141  // Convert to an unsigned.
142  sscanf(gidText, "%u", &gid);
143 
144  // Find the tileset index.
145  const int tilesetIndex = map->FindTilesetIndex(gid);
146  if (tilesetIndex != -1)
147  {
148  // If valid, set up the map tile with the tileset.
149  const Tmx::Tileset* tileset = map->GetTileset(tilesetIndex);
150  tile_map[tileCount] = MapTile(gid, tileset->GetFirstGid(), tilesetIndex);
151  }
152  else
153  {
154  // Otherwise, make it null.
155  tile_map[tileCount] = MapTile(gid, 0, -1);
156  }
157 
158  //tileNode = dataNode->IterateChildren("tile", tileNode); FIXME MAYBE
159  tileNode = tileNode->NextSiblingElement("tile");
160  tileCount++;
